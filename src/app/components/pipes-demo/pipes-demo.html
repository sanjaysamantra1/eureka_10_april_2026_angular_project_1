let's Learn Angular Pipes

uppercase, lowercase, titlecase pipe::
UserName: {{userName}}
UserName with uppercase pipe: {{userName | uppercase}}
UserName with lowercase pipe: {{userName | lowercase}}
UserName with titlecase pipe: {{userName | titlecase}}
Currency Pipe::
Salary: {{mySal | currency}}
Salary: {{mySal | currency : 'INR'}}
Salary: {{mySal | currency : '€'}}
Salary: {{mySal | currency : '¥'}}
Salary: {{mySal | currency : 'USD' : 'symbol'}}
Salary: {{mySal | currency : 'USD' : 'code'}}
Date Pipe::
{{dateObj}}
{{ dateObj | date }}
{{ dateObj | date :'short'}}
{{ dateObj | date :'medium'}}
{{ dateObj | date :'long'}}
{{ dateObj | date :'full'}}
==================
{{ dateObj | date : 'shortDate' }}
{{ dateObj | date : 'mediumDate' }}
{{ dateObj | date : 'longDate' }}
{{ dateObj | date : 'fullDate' }}
=================
{{ dateObj | date : 'shortTime' }}
{{ dateObj | date : 'mediumTime' }}
{{ dateObj | date : 'longTime' }}
{{ dateObj | date : 'fullTime' }}
==============
{{ dateObj | date : 'dd-MMM-yyyy' }}
{{ dateObj | date : 'hh:mm:ss a' }}
{{ dateObj | date: "fullTime":"UTC" }}
{{ dateObj | date: "fullTime":"IST" }}
{{ dateObj | date: "EEEE" }}
Number/Decimal Pipe
{{ 100000000 | number }}
{{ 12.111222333 | number: "3.2-5" }}
{{ 12.1 | number: "3.2-5" }}
{{ 0.5 | number:'3.2-5'}}
Percent Pipe
{{ 2.5 | percent}}
{{ 2.5 | percent:'2.2-5'}}
{{ 0.024 | percent}}

{{'Angular is a Framework' | slice : 0 : -4}}
